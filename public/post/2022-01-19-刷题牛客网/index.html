class Bonus {
public:
    int getMost(vector<vector<int> > board) {
        // write code here
        // dp[i][j]: 位于(i, j)获得的最大价值
        // dp[i][j] = max(dp[i][j-1], dp[i-1][j]) + board[i][j];
        int n = 6, m = 6;
        vector<vector<int>> dp(n, vector<int>(m, 0));
        dp[0][0] = board[0][0];
        for (int i=1; i<n; i++) {
            dp[i][0] = dp[i-1][0] + board[i][0];
        }
        for (int j=1; j<m; j++) {
            dp[0][j] = dp[0][j-1] + board[0][j];
        }
        for (int i=1; i<n; i++) {
            for (int j=1; j<m; j++) {
                dp[i][j] = max(dp[i][j-1], dp[i-1][j]) + board[i][j];
            }
        }
        return dp[n-1][m-1];
    }
};
